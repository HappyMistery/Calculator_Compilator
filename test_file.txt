// Basic integer operations
7 + 3               // 10
10 - 4              // 6
3 * 4               // 12
5 % 2               // 1
2 ** 3              // 8

// Complex integer operations
(7 + 3) * 2 - 5    // 15
(10 - 4) / 2 + 8 * 3 // 26
3 * (4 + 2 ** 3)   // 24
5 % 2 + 3 * 2 - 4 / 2 // 5

// Basic float operations
7.5 + 2.5          // 10.0
10.2 - 4.1         // 6.1
3.0 * 4.0          // 12.0
2 / 4               // 0.5
9.0 / 3.0          // 3.0
8.5 % 3.0          // 2.5
2.5 ** 2.0         // 6.25

// Complex float operations
(7.5 + 2.5) / 2.0 - 1.5 * 3 // 0.0
(10.2 - 4.1) * 2.0 + 1.0 / 2.0 // 12.2
3.0 * (4.0 + 2.5 ** 2.0) // 27.5
(8.5 % 3.2) + 4.2 * 1.5 - 3.0 / 1.5 // 8.05

// Basic trigonometric operations (int and float)

tan(0)            // 0

tan(PI / 4)      // 1

cos(PI / 3)      // 0.5
tan(PI / 3)      // 1.732051

// Basic mixed operations (int and float)

// Basic mixed operations (int and trig)

// Basic mixed operations (float and trig)
7.5 + sin(PI / 6) // 8.0
10.5 - cos(PI / 4) // 9.79289
3.5 * tan(PI / 3) // 6.06218

// Complex mixed operations (int and float)
(7 + 2.5) * 3 - 4 / 2 + 1.5 // 24.5
10 - (3.5 * 2) + 8**2 - 4 / 3.0 // 62.333
(3 * 2.0 + 1.5**2) - 7 / 3 // 5.75
(5%2.5 + 4.5) * 3 - 2 / 1.0 // 12.0

// Complex mixed operations (int and trig)
(30 + sin(30)) * 2 - 4 / 2 // 61.999
10 - (3 * cos(30)) + 5 - 3 // 11.464

// Complex mixed operations (float and trig)
(7.5 + sin(PI / 4)) * 2 - 1.5 // 14.414
10.2 - (3 * cos(PI / 4)) + 1.5 // 11.914
