// Basic integer operations
7 + 3               // 10
10 - 4              // 6
3 * 4               // 12
5 % 2               // 1
2 ** 3              // 8

// Complex integer operations
(7 + 3) * 2 - 5    // 15
(10 - 4) / 2 + 8 * 3 // 26
3 * (4 + 2 ** 3)   // 24
5 % 2 + 3 * 2 - LEN("hola") / 2 // 5

// Basic float operations
7.5 + 2.5          // 10.0
10.2 - 4.1         // 6.1
3.0 * 4.0          // 12.0
2 / 4               // 0.5
9.0 / 3.0          // 3.0
8.5 % 3.0          // 2.5
2.5 ** 2.0         // 6.25
3 ** 2.5         // 15.5885

// Complex mixed operations (int and float)
(7 + 2.5) * 3 - 4 / 2 + 1.5 // 24.5
10 - (3.5 * 2) + LEN("perfecto")**2 - 4 / 3.0 // 62.333
(3 * 2.0 + 1.5**2) - 7 / 3 // 5.75
(5%2.5 + 4.5) * 3 - 2 / 1.0 // 12.0

// Basic trigonometric operations (int and float)
sin(0)            // 0
cos(0)            // 1
tan(0)            // 0
sin(PI / 4)      // 0.707107
cos(PI / 4)      // 0.707107
tan(PI / 4)      // 1

// Complex trigonometric operations (int and float)
cos(PI / 3)      // 0.5
tan(PI / 3)      // 1.732051

// Basic mixed operations (int and trig)
LEN("Las estrellas brillan de noche") + sin(30)     // 30.499
45 - cos(30)     // 44.134
60 * tan(30)     // 34.641

// Basic mixed operations (float and trig)
7.5 + sin(PI / 6) // 8.0
10.5 - cos(PI / 4) // 9.79289
3.5 * tan(PI / 3) // 6.06218

// Complex mixed operations (int and trig)
(30 + sin(30)) * 2 - 4 / 2 // 61.999
10 - (3 * cos(30)) + 5 - 3 // 11.464

// Complex mixed operations (float and trig)
(7.5 + sin(PI / 4)) * 2 - 1.5 // 14.414
10.2 - (3 * cos(PI / LEN("duck"))) + 1.5 // 11.914

// Basic constant operations (PI and E)
PI                 // 3.14159
E                  // 2.71828
PI * E             // 8.53973

// Complex constant operations (PI and E)
PI**4        // 9.8696
E**2         // 7.38906
sin(PI)           // 0
cos(E)            // -0.91173

// Basic mixed operations (int, float, trig, PI, and E)
8 / PI            // 2.54648
sin(PI / 2) + 1   // 2

// Complex mixed operations (int, float, trig, PI, and E)
(7 + E) * 3 - sin(PI) // 20.155
10 - (3.5 * PI) + E**2 - LEN("Piernas al fallo") / 3.0 // 1.080

/* 
String operations using
string concat (+), 
SUBSTR and both of them 
at the same time
*/
"Hello, " + "World!"                      // Basic string concatenation
"Good " + "Morning " + "Everyone"           // Multiple strings concatenation
"The result is: " + "42"                  // Concatenating a string with a numeric string
SUBSTR("Hello professor" 6 10)           // Basic substringing
"A boolean can be" + " either " + true + " or " + false     //String concatenation with a boolean
"Pi is approximately: " + "3.14"          // String with numeric value
"C" + "++" + " Programming"                  // Concatenating without spaces
"Concatenation of " + "this" + " and " + "that" // Concatenating with spaces
"Hello " + "there, " + "how " + "are " + "you?" // Greeting concatenation
"String " + "operations " + "are " + "fun!"   // Fun statement concatenation
SUBSTR(("xd" + 123) 2 3)                    // Substring of a concatenation
"To say that 1 == 1 is " + true and (1==1)    // String concat with boolean operations
"The quick " + "brown " + "fox " + "jumps"    // Concatenation of a phrase
"Concatenating " + "strings " + "in C"       // Concatenating in C
